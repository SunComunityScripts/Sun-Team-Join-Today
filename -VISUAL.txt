	local pauseTime = 0.5 -- pausa no fim da rota√ß√£o
	
	-- Fun√ß√£o para girar a ImageLabel
	local function rotateOnce()
		-- Criar Tween para girar 360¬∞
		local tweenInfo = TweenInfo.new(
			rotationTime,
			Enum.EasingStyle.Sine, -- anima√ß√£o suave
			Enum.EasingDirection.InOut
		)
		local goal = {Rotation = imageLabel.Rotation + 360}
		local tween = TweenService:Create(imageLabel, tweenInfo, goal)
	
		tween:Play()
		tween.Completed:Wait()
	end
	
	-- Loop infinito
	while true do
		rotateOnce()
		wait(pauseTime) -- pausa de 0,5s
		-- Girar de volta suavemente
		local tweenInfo = TweenInfo.new(
			rotationTime,
			Enum.EasingStyle.Sine,
			Enum.EasingDirection.InOut
		)
		local goal = {Rotation = imageLabel.Rotation - 360}
		local tween = TweenService:Create(imageLabel, tweenInfo, goal)
		tween:Play()
		tween.Completed:Wait()
		wait(pauseTime)
	end
	
end;
